# usePageSize

`usePageSize` provides getter/setter for the resultsPerPage param.

```js
import { Results } from '@sajari/react-search-ui';
import { Pagination, Select, Input } from '@sajari/react-components';
import { usePageSize, useSearch, SearchContextProvider, Pipeline, Variables } from '@sajari/react-hooks';
```

## Usage

```jsx
function Example() {
  const pipeline = new Pipeline(
    {
      project: '1594153711901724220',
      collection: 'bestbuy',
      endpoint: 'https://jsonapi-us-valkyrie.sajari.net',
    },
    'query',
  );

  const variables = new Variables();

  const SearchPlayground = React.memo(() => {
    const { search, results } = useSearch();
    const { pageSize, setPageSize } = usePageSize();
    const { query } = useQuery();

    return (
      <div className="flex flex-col space-y-4">
        <div className="flex space-x-4">
          <div className="w-3/5">
            <Input label="Search something" value={query} onChange={(value) => search(value)} />
          </div>
          <div className="w-2/5">
            <Select value={`${pageSize}`} onChange={(e) => setPageSize(parseInt(e.target.value), 10)}>
              <option value="3">3</option>
              <option value="5">5</option>
              <option value="10">10</option>
              <option value="15">15</option>
            </Select>
          </div>
        </div>
        {results ? <Results /> : null}
      </div>
    );
  });

  return (
    <SearchContextProvider
      search={{
        pipeline,
        variables,
        fields: {
          category: (data) => data.level4 || data.level3 || data.level2 || data.level1 || data.brand,
          title: 'name',
        },
      }}
    >
      <SearchPlayground />
    </SearchContextProvider>
  );
}
```

## Anatomy

`usePageSize` returns an object containing the following properties:

| Name          | Type                     | Default | Description                             |
| ------------- | ------------------------ | ------- | --------------------------------------- |
| `pageSize`    | `number`                 |         | The number of result per page           |
| `setPageSize` | `(size: number) => void` |         | Method to set number of result per page |
