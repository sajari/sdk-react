# Click Tracking

```js
import { Results } from '@sajari/react-search-ui';
import { Input } from '@sajari/react-components';
import { SearchProvider, Pipeline, Variables, ClickTracking } from '@sajari/react-hooks';
```

## Usage

When `ClickTracking` is used, a token object containing a link will be returned for each result item. Details can be found [here](https://www.sajari.com/docs/user-guide/analytics/click-tracking/)

```jsx
function Example() {
  const pipeline = new Pipeline(
    {
      account: '1594153711901724220',
      collection: 'bestbuy',
      endpoint: 'https://jsonapi-us-valkyrie.sajari.net',
    },
    'query',
    new ClickTracking(),
  );

  const SearchPlayground = React.memo(() => (
    <div className="flex flex-col space-y-6">
      <Input />
      <Results />
    </div>
  ));

  return (
    <SearchProvider
      search={{
        pipeline,
        fields: {
          title: 'name',
          subtitle: (data) => data.level4 || data.level3 || data.level2 || data.level1 || data.brand,
        },
      }}
    >
      <SearchPlayground />
    </SearchProvider>
  );
}
```
