# NoTracking

```js
import { Results } from '@sajari/react-search-ui';
import { Input } from '@sajari/react-components';
import { SearchContextProvider, Pipeline, Variables, ClickTracking } from '@sajari/react-hooks';
```

## Usage

This is the default when creating a new Pipeline instance.

```jsx
function Example() {
  const pipeline = new Pipeline(
    {
      project: '1594153711901724220',
      collection: 'bestbuy',
      endpoint: '//jsonapi-us-valkyrie.sajari.net',
    },
    'query',
  );

  const variables = new Variables();

  const SearchPlayground = React.memo(() => (
    <div className="flex flex-col space-y-4">
      <div>
        <Input label="Search something" />
      </div>
      <Results />
    </div>
  ));

  return (
    <SearchContextProvider search={{ pipeline, variables, fields: { title: 'name', subtitle: 'brand' } }}>
      <SearchPlayground />
    </SearchContextProvider>
  );
}
```
