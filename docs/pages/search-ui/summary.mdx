# Summary

Summary is a component used to display display the number of results and the latency from a search query.

```js
import { Summary } from '@sajari/react-search-ui';
```

## Usage

```jsx
function Example() {
  useSearch({ queryOverride: 'iphone' });

  return <Summary />;
}
```

### Override a query via autocomplete

```jsx
function Example() {
  const { search, searchInstant } = useSearch();
  const { query } = useQuery();

  return (
    <div className="flex flex-col space-y-3">
      <Combobox
        value={query}
        onChange={(v) => {
          search(v);
          searchInstant(v); // Triggers autocomplete pipeline
        }}
      />
      <Summary />
    </div>
  );
}
```

### Hide the query override

```jsx
function Example() {
  const { search, searchInstant } = useSearch({ allowEmptySearch: false });
  const { query } = useQuery();

  return (
    <div className="flex flex-col space-y-3">
      <Combobox
        value={query}
        onChange={(v) => {
          search(v);
          searchInstant(v); // Triggers autocomplete pipeline
        }}
      />
      <Summary showOverride={false} />
    </div>
  );
}
```
