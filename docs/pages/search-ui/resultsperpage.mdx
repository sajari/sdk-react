# ResultsPerPage

The ResultsPerPage component renders a select component to allow setting results displayed per page.

```js
import { Input, Results, ResultsPerPage } from '@sajari/react-search-ui';
import { SearchProvider, Pipeline } from '@sajari/react-hooks';
```

## Usage

```jsx
function Example() {
  const pipeline = new Pipeline(
    {
      account: '1594153711901724220',
      collection: 'bestbuy',
      endpoint: 'https://jsonapi-us-valkyrie.sajari.net',
    },
    'query',
  );

  const SearchPlayground = React.memo(() => (
    <div className="space-y-6">
      <div className="flex items-center space-x-4">
        <div className="flex-1">
          <Input />
        </div>

        <ResultsPerPage />
      </div>

      <Results />

      <Pagination />
    </div>
  ));

  return (
    <SearchProvider
      search={{
        pipeline,
        fields: {
          title: 'name',
          subtitle: (data) => data.level4 || data.level3 || data.level2 || data.level1 || data.brand,
        },
      }}
      searchOnLoad
    >
      <SearchPlayground />
    </SearchProvider>
  );
}
```

## Props

| Name    | Type                 | Default               | Description                   |
| ------- | -------------------- | --------------------- | ----------------------------- |
| `label` | `string`             | `Show`                |                               |
| `sizes` | `Array<number>`      | [10, 20, 35, 50, 100] |                               |
| `size`  | `sm` \| `md` \| `lg` | `md`                  | Sets the text and input size. |
