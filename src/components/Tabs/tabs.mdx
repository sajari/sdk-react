# Tabs

## Usage

<ExampleContainer>
  <Tabs
    filter={categoryFilter()}
    tabs={[
      { id: "All", display: "All" },
      { id: "Blog", display: "Blog" }
    ]}
  />
</ExampleContainer>

```javascript
import { Filter, Tabs } from "@sajari/sdk-react";

const filter = new Filter(
  {
    All: "",
    Blog: "dir1='blog'"
  },
  ["All"]
);

<Tabs
  filter={filter}
  tabs={[
    { id: "All", display: "All" },
    { id: "Blog", display: "Blog" }
  ]}
/>;
```
