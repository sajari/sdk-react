# RangeSlider

```javascript
// Import the RangeSlider component
import { RangeSlider } from "@sajari/sdk-react";
```

## Usage

<Playground>
{() => {
    const rangeFilter = new RangeFilter('range', [0, 500]);
    rangeFilter.set(200,300);
    const [filterOutput, setFilterOutput ] = React.useState(rangeFilter.filter());

    const removeListener = rangeFilter.listen(EVENT_SELECTION_UPDATED, () => {
      setFilterOutput(rangeFilter.filter())
    });

    return (
      <>
        <RangeSlider
          filter={rangeFilter}
          pushable={50}
          allowCross={false}
        />
        <pre>"{filterOutput}"</pre>
      </>
    )

}}

</Playground>
