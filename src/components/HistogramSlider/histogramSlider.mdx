# Histogram Slider

```javascript

import { HistogramSlider } from "@sajari/sdk-react";
```

## Usage

### With default width & height

<Playground>
  <HistogramSlider
    min={0}
    max={1000}
    step={1}
    value={[0, 1000]}
    distance={50}
    data={sampleData}
  />
</Playground>

### Custom width, height & colors

<Playground>
  <HistogramSlider
    colors={{
      in: "#2d8be8",
      out: "#b6d2ef"
    }}
    min={162}
    max={14000}
    step={2}
    value={[162, 14000]}
    distance={1000}
    widthPx={200}
    heightPx={100}
    data={sampleData}
  />
</Playground>

### Custom Info Render & opApply call

<Playground>
  <HistogramSlider    
    min={0}
    max={1000}
    step={1}
    value={[0, 1000]}
    distance={50}
    data={sampleData}
    InfoRenderComponent={({ value }) => (
      <div style={{ marginBottom: "10px" }}>
        From <b>{value[0]}</b> to <b>{value[1]}</b>
      </div>
    )}
    onApply={value => {
      if(value[0] !== 0 || value[1] !== 1000){
        alert(value);
      }
    }}

/>

</Playground>
